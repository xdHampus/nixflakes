        int index = i;
        for (int j = i + 1; j < arr.length; j++) {
            if (arr[j] < arr[index]) {
                index = j; //searching for lowest index
            }
        }
        int smallerNumber = arr[index]; 
        arr[index] = arr[i];
        arr[i] = smallerNumber;
    }
